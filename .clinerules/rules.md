class GenerationConfig(BaseModel):

class RootConfig(BaseModel):

    git: GitConfig = Field(default_factory=GitConfig)

# Add this content to cli.py

import sys
import os
import importlib

def main():
    if len(sys.argv) < 2:
        print("Usage: python cli.py <command> [args...]")
        sys.exit(1)

    command_name = sys.argv[1]
    
    try:
        # Dynamically import the command module
        command_module = importlib.import_module(f"commands.{command_name}")
        
        # Assume each command module has a 'run' function
        if hasattr(command_module, "run") and callable(command_module.run):
            command_module.run(sys.argv[2:])
        else:

            sys.exit(1)
    except ModuleNotFoundError:
        print(f"Error: Unknown command '{command_name}'.")
        sys.exit(1)
    except Exception as e:

        sys.exit(1)
